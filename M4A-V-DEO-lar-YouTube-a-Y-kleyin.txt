    function log(msg) {
      document.getElementById('log').textContent += msg + '\n';

    document.getElementById('uploadYT').onclick = async () => {
      const fileInput = document.getElementById('file');
      const thumbnailInput = document.getElementById('thumbnail');
      const title = document.getElementById('title').value || 'Şarkı Adı (Official Music)';
      const description = document.getElementById('desc').value || '';
      const privacy = document.getElementById('privacy').value;
      if (!fileInput.files.length) {
        alert("Lütfen bir video dosyası seçin.");
        return;
      }
      const file = fileInput.files[0];
      log("YouTube'a Video Yükleniyor...");
      const metadata = {
        snippet: {
          title: title,
          description: description
        },
        status: {
          privacyStatus: privacy
        }
      };
      const form = new FormData();
      form.append('snippet', new Blob([JSON.stringify(metadata)], {
        type: 'application/json'
      }));
      form.append('video', file);
      try {
        const response = await fetch('https://www.googleapis.com/upload/youtube/v3/videos?part=snippet,status', {
          method: 'POST',
          headers: {
            'Authorization': `Bearer ${accessToken}`
          },
          body: form
        });
        const result = await response.json();
        if (response.ok) {
          log("YouTube'a Video Yüklendi. Video ID: " + result.id);
          updateShareLinks(result.id);
          if (thumbnailInput.files.length) {
            await uploadThumbnail(result.id, thumbnailInput.files[0]);
          }
        } else {
          log("Hata: " + JSON.stringify(result));
        }
      } catch (err) {
        log("Yükleme hatası: " + err.message);
      }
    };
    async function uploadThumbnail(videoId, thumbnailFile) {
      log("YouTube'a Kapak Yükleniyor...");
      const form = new FormData();
      form.append('media', thumbnailFile);
      const res = await fetch(`https://www.googleapis.com/upload/youtube/v3/thumbnails/set?videoId=${videoId}`, {
        method: 'POST',
        headers: {
          'Authorization': `Bearer ${accessToken}`
        },
        body: form
      });
      const result = await res.json();
      if (res.ok) {
        log("YouTube'a Kapak Yüklendi.");
      } else {
        log("YouTube'a Kapak Yükleme hatası: " + JSON.stringify(result));
      }
    }

    function updateShareLinks(videoId) {
      const videoUrl = `https://youtu.be/${videoId}`;

      document.getElementById('facebookBtn').href = `https://www.facebook.com/sharer/sharer.php?u=${encodeURIComponent(videoUrl)}`;
      document.getElementById('twitterBtn').href = `https://twitter.com/intent/tweet?url=${encodeURIComponent(videoUrl)}`;
      document.getElementById('telegramBtn').href = `https://t.me/share/url?url=${encodeURIComponent(videoUrl)}`;
      document.getElementById('emailBtn').href = `mailto:?subject=Videoyu İzle&body=${encodeURIComponent(videoUrl)}`;
      window.currentVideoUrl = videoUrl;
    }

    function setupShareLinks() {
      const defaultUrl = window.location.href;

      window.currentVideoUrl = defaultUrl;
    }

    function copyVideoLink() {
      if (window.currentVideoUrl) {
        navigator.clipboard.writeText(window.currentVideoUrl)
          .then(() => alert('Video bağlantısı kopyalandı!'))
          .catch(() => alert('Kopyalama başarısız oldu.'));
      }
    }

    function copyEmbedCode() {
      if (window.currentVideoUrl) {
        const videoId = window.currentVideoUrl.split('/').pop();
        const embedCode = `<iframe width="560" height="315" src="https://www.youtube.com/embed/${videoId}" frameborder="0" allowfullscreen></iframe>`;
        navigator.clipboard.writeText(embedCode)
          .then(() => alert('Embed kodu kopyalandı!'))
          .catch(() => alert('Kopyalama başarısız oldu.'));
      }
    }
